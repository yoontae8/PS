

class Solution(object):
    def twoSum(self, nums, target):
        """
        :type nums: List[int]
        :type target: int
        :rtype: List[int]
        """
        min_val = min(nums)
        max_val = max(nums)
        for i in range(len(nums)-1): 
            left = nums[i]
            if left + min_val > target : 
                continue 
            if left + max_val < target : 
                continue 
            for j in range(len(nums)-(i+1)): 
                right = nums[i+1+j]
                if left+right == target : 
                    return [i, i+1+j]
